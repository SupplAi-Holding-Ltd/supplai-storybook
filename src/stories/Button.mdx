<Meta of={ButtonStories} />

export const ButtonConfigurator = () => {
  const [variant, setVariant] = React.useState("filled");
  const [intent,  setIntent]  = React.useState("primary");
  const [size,    setSize]    = React.useState("md");
  const [withLeading,  setWithLeading]  = React.useState(false);
  const [withTrailing, setWithTrailing] = React.useState(false);
  const [isDisabled,   setIsDisabled]   = React.useState(false);

  const variants = ["filled", "outlined", "tonal", "ghost"];
  const intents  = ["primary", "success", "warning", "error", "neutral"];
  const sizes    = ["xs", "sm", "md", "lg", "xl"];

  const intentColors = {
    primary: "#4169E1", success: "#2D9C93", warning: "#FF9F05",
    error: "#D13145", neutral: "#6B7280",
  };

  return (
    <div className="bc-wrap">
      {/* Preview */}
      <div className="bc-preview">
        <Button
          variant={variant}
          intent={intent}
          size={size}
          label="Button"
          leadingIcon={withLeading ? <PlusIcon /> : undefined}
          trailingIcon={withTrailing ? <ArrowRightIcon /> : undefined}
          disabled={isDisabled}
        />
      </div>

      {/* Controls */}
      <div className="bc-controls">
        <div className="bc-control-group">
          <span className="bc-ctrl-label">Variant</span>
          <div className="bc-chips">
            {variants.map(v => (
              <button key={v} className={`bc-chip ${variant === v ? "bc-chip--on" : ""}`} onClick={() => setVariant(v)}>{v}</button>
            ))}
          </div>
        </div>

        <div className="bc-control-group">
          <span className="bc-ctrl-label">Intent</span>
          <div className="bc-chips">
            {intents.map(i => (
              <button
                key={i}
                className={`bc-chip ${intent === i ? "bc-chip--on" : ""}`}
                style={intent === i ? { background: intentColors[i], borderColor: intentColors[i], color: "#fff" } : {}}
                onClick={() => setIntent(i)}
              >{i}</button>
            ))}
          </div>
        </div>

        <div className="bc-control-group">
          <span className="bc-ctrl-label">Size</span>
          <div className="bc-chips">
            {sizes.map(s => (
              <button key={s} className={`bc-chip ${size === s ? "bc-chip--on" : ""}`} onClick={() => setSize(s)}>{s}</button>
            ))}
          </div>
        </div>

        <div className="bc-control-group">
          <span className="bc-ctrl-label">Options</span>
          <div className="bc-chips">
            <button className={`bc-chip ${withLeading  ? "bc-chip--on" : ""}`} onClick={() => setWithLeading(v => !v)}>Leading icon</button>
            <button className={`bc-chip ${withTrailing ? "bc-chip--on" : ""}`} onClick={() => setWithTrailing(v => !v)}>Trailing icon</button>
            <button className={`bc-chip ${isDisabled   ? "bc-chip--on" : ""}`} onClick={() => setIsDisabled(v => !v)}>Disabled</button>
          </div>
        </div>
      </div>
    </div>
  );
};
