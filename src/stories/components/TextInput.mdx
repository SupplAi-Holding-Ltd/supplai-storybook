import React from "react";

<Meta title="Components/Form Methods/Text Input" />

export const TextInputPlayground = () => {
  const [value, setValue]       = React.useState("");
  const [forceState, setForce]  = React.useState("default");
  const [withIcon, setWithIcon] = React.useState(false);

  const derivedState = forceState !== "default" ? forceState
    : value.length === 0 ? "default"
    : "filled";

  const borderColor = {
    default:  "#D1D5DB",
    hover:    "#9CA3AF",
    focused:  "#4169E1",
    filled:   "#D1D5DB",
    active:   "#4169E1",
    error:    "#D13145",
    success:  "#D1D5DB",
  }[derivedState] || "#D1D5DB";

  const shadow = (derivedState === "focused" || derivedState === "active")
    ? "0 0 0 3px rgba(65,105,225,0.12)"
    : derivedState === "error"
    ? "0 0 0 3px rgba(209,49,69,0.10)"
    : "none";

  const bg = derivedState === "error" ? "#FFF5F6" : "#FFFFFF";
  const textColor = derivedState === "error" ? "#D13145" : "#111625";

  const stateOptions = ["default","hover","focused","active","error","success"];

  return (
    <div className="tip-wrap">
      {/* Input preview */}
      <div className="tip-preview">
        <label className="tip-label">Label</label>
        <div className="tip-input-row">
          {withIcon && (
            <svg className="tip-icon" width="16" height="16" viewBox="0 0 16 16" fill="none">
              <rect x="1" y="3" width="14" height="10" rx="2" stroke="#9CA3AF" strokeWidth="1.4"/>
              <path d="M1 6l7 4 7-4" stroke="#9CA3AF" strokeWidth="1.4"/>
            </svg>
          )}
          <input
            className="tip-input"
            type="text"
            placeholder="Start typing…"
            value={value}
            onChange={e => { setValue(e.target.value); setForce("default"); }}
            style={{
              borderColor,
              boxShadow: shadow,
              background: bg,
              color: textColor,
              paddingLeft: withIcon ? "34px" : "12px",
            }}
          />
        </div>
        {derivedState === "error" && (
          <span className="tip-helper tip-helper--error">
            <svg width="13" height="13" viewBox="0 0 16 16" fill="none"><circle cx="8" cy="8" r="7" stroke="#D13145" strokeWidth="1.5"/><path d="M8 5v3.5" stroke="#D13145" strokeWidth="1.5" strokeLinecap="round"/><circle cx="8" cy="11" r="0.75" fill="#D13145"/></svg>
            This field has an error
          </span>
        )}
        {derivedState === "success" && (
          <span className="tip-helper tip-helper--success">
            <svg width="13" height="13" viewBox="0 0 16 16" fill="none"><circle cx="8" cy="8" r="7" stroke="#2D9C93" strokeWidth="1.5"/><path d="M5 8.5l2 2 4-4" stroke="#2D9C93" strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round"/></svg>
            Looks good!
          </span>
        )}
      </div>

      {/* Controls */}
      <div className="tip-controls">
        <div className="tip-ctrl-group">
          <span className="tip-ctrl-label">Force state</span>
          <div className="tip-chips">
            {stateOptions.map(s => (
              <button
                key={s}
                className={`tip-chip ${forceState === s ? "tip-chip--on" : ""}`}
                onClick={() => setForce(s)}
              >{s}</button>
            ))}
          </div>
        </div>
        <div className="tip-ctrl-group">
          <span className="tip-ctrl-label">Options</span>
          <div className="tip-chips">
            <button className={`tip-chip ${withIcon ? "tip-chip--on" : ""}`} onClick={() => setWithIcon(v => !v)}>
              With icon
            </button>
            <button className="tip-chip" onClick={() => { setValue(""); setForce("default"); }}>
              Clear
            </button>
          </div>
        </div>
        <div className="tip-state-display">
          Current state: <strong>{derivedState}</strong>
        </div>
      </div>
    </div>
  );
};
