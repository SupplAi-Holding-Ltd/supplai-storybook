export const TOGGLE_CODE = `// Toggle · Supplai Design System
import React, { useState } from 'react';

function Toggle({ checked = false, onChange, disabled = false, label }) {
  const handleKey = e => {
    if (!disabled && (e.key === ' ' || e.key === 'Enter')) {
      e.preventDefault();
      onChange?.(!checked);
    }
  };

  return (
    <label
      style={{
        display: 'inline-flex',
        alignItems: 'center',
        gap: 10,
        cursor: disabled ? 'not-allowed' : 'pointer',
        opacity: disabled ? 0.45 : 1,
      }}
    >
      <span
        role="switch"
        aria-checked={checked}
        tabIndex={disabled ? -1 : 0}
        onClick={() => !disabled && onChange?.(!checked)}
        onKeyDown={handleKey}
        style={{
          position: 'relative',
          display: 'inline-block',
          width: 44,
          height: 24,
          borderRadius: 99,
          background: checked ? '#4169E1' : '#D1D5DB',
          transition: 'background 0.2s',
          cursor: disabled ? 'not-allowed' : 'pointer',
          flexShrink: 0,
          outline: 'none',
        }}
      >
        <span
          style={{
            position: 'absolute',
            top: 2,
            left: checked ? 22 : 2,
            width: 20,
            height: 20,
            borderRadius: '50%',
            background: '#FFFFFF',
            boxShadow: '0 1px 4px rgba(0,0,0,0.2)',
            transition: 'left 0.2s cubic-bezier(0.4, 0, 0.2, 1)',
          }}
        />
      </span>
      {label && (
        <span style={{ fontSize: 13, fontWeight: 500, color: disabled ? '#9CA3AF' : '#374151' }}>
          {label}
        </span>
      )}
    </label>
  );
}

// ── Usage ─────────────────────────────────────────────────────────────
function App() {
  const [on, setOn] = useState(false);

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      {/* Basic */}
      <Toggle checked={on} onChange={setOn} />

      {/* With label */}
      <Toggle checked={on} onChange={setOn} label={on ? 'On' : 'Off'} />

      {/* Disabled */}
      <Toggle checked={false} disabled label="Disabled off" />
      <Toggle checked={true}  disabled label="Disabled on" />
    </div>
  );
}`;
