import React, { useState } from 'react';

function PasswordInput({ label, placeholder, error, helperText }) {
  const [value, setValue]     = useState('');
  const [focused, setFocused] = useState(false);
  const [visible, setVisible] = useState(false);

  const getStrength = pwd => {
    let s = 0;
    if (pwd.length >= 8)           s++;
    if (/[A-Z]/.test(pwd))         s++;
    if (/[0-9]/.test(pwd))         s++;
    if (/[^A-Za-z0-9]/.test(pwd)) s++;
    return s;
  };

  const strength = getStrength(value);
  const strengthColors = ['', '#D13145', '#FF9F05', '#4169E1', '#2D9C93'];
  const strengthLabels = ['', 'Weak', 'Fair', 'Good', 'Strong'];

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
      {label && (
        <label style={{ fontSize: 13, fontWeight: 600, color: '#374151' }}>
          {label}
        </label>
      )}

      <div style={{ position: 'relative', display: 'inline-flex', alignItems: 'center' }}>
        {/* Lock icon */}
        <LockIcon style={{ position: 'absolute', left: 10, pointerEvents: 'none' }} />

        <input
          type={visible ? 'text' : 'password'}
          value={value}
          placeholder={placeholder}
          onChange={e => setValue(e.target.value)}
          onFocus={() => setFocused(true)}
          onBlur={() => setFocused(false)}
          style={{
            height: 36,
            padding: '8px 36px 8px 32px',
            borderRadius: 6,
            borderStyle: 'solid',
            borderWidth: 1,
            borderColor: error   ? '#D13145'
                       : focused ? '#4169E1'
                       :           '#D1D5DB',
            boxShadow: focused && !error
              ? '0 0 0 3px rgba(65,105,225,0.12)'
              : 'none',
            background: error ? '#FFF5F6' : '#FFFFFF',
            color: error ? '#D13145' : '#111625',
            fontSize: 13,
            fontFamily: 'Manrope, sans-serif',
            outline: 'none',
            width: '100%',
          }}
        />

        {/* Eye toggle */}
        <button
          onClick={() => setVisible(v => !v)}
          style={{ position: 'absolute', right: 10, background: 'none', border: 'none', cursor: 'pointer', padding: 0 }}
        >
          <EyeIcon open={visible} />
        </button>
      </div>

      {/* Strength meter */}
      {value.length > 0 && (
        <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
          {[1,2,3,4].map(i => (
            <div key={i} style={{
              flex: 1, height: 4, borderRadius: 99,
              background: i <= strength ? strengthColors[strength] : '#E5E7EB',
              transition: 'background 0.2s',
            }} />
          ))}
          <span style={{ fontSize: 12, fontWeight: 600, color: strengthColors[strength], flexShrink: 0 }}>
            {strengthLabels[strength]}
          </span>
        </div>
      )}

      {error && helperText && (
        <span style={{ fontSize: 12, color: '#D13145' }}>{helperText}</span>
      )}
    </div>
  );
}

    : value.length === 0 ? "default"
    : "filled";
